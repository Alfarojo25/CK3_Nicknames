ac_nickname_toggle_debug = {
	desc = ac_nickname_toggle_debug_desc
	selection_tooltip = ac_nickname_toggle_debug_tooltip
	is_shown = {
		always = yes
	}
	is_valid = {
		is_ruler = yes
	}
	effect = {
		if = {
			limit = {
				has_character_flag = ac_nickname_debug_mode
			}
			remove_character_flag = ac_nickname_debug_mode
			custom_tooltip = ac_nickname_debug_mode_off_tt
		}
		else = {
			add_character_flag = ac_nickname_debug_mode
			custom_tooltip = ac_nickname_debug_mode_on_tt
		}
	}
	ai_will_do = {
		base = 0
	}
}

ac_nickname_give_all = {
	desc = ac_nickname_give_all_desc
	selection_tooltip = ac_nickname_give_all_tooltip
	is_shown = {
		has_character_flag = ac_nickname_debug_mode
	}
	is_valid = {
		is_ruler = yes
	}
	effect = {
		custom_tooltip = ac_nickname_give_all_tt
		hidden_effect = {
			every_living_character = {
				limit = {
					NOT = {
						has_nickname = yes
					}
					age >= 16
				}
				trigger_event = {
					id = ac_nickname.1001
					days = 1
				}
			}
		}
	}
	cost = {
		prestige = 0
		piety = 0
	}
	ai_will_do = {
		base = 0
	}
}
